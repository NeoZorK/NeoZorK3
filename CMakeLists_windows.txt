cmake_minimum_required(VERSION 3.20)
project(SolanaArbitrageBot VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Build type
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

# Windows-specific settings
if(WIN32)
    # Windows compiler flags
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W4 /O2")
    set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} /Zi /Od /DDEBUG")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /O2 /DNDEBUG")
    
    # Windows-specific definitions
    add_definitions(-DWIN32_LEAN_AND_MEAN)
    add_definitions(-DNOMINMAX)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
    
    # Set Windows subsystem
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:CONSOLE")
else()
    # Unix-like compiler flags
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -O3")
    set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g -O0 -DDEBUG")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3 -DNDEBUG")
endif()

# Find required packages
find_package(Boost REQUIRED COMPONENTS system thread)
find_package(OpenSSL REQUIRED)
find_package(Threads REQUIRED)

# For Windows, we need to find nlohmann/json manually
if(WIN32)
    find_path(NLOHMANN_JSON_INCLUDE_DIR nlohmann/json.hpp
        PATHS 
        ${CMAKE_SOURCE_DIR}/external
        ${CMAKE_SOURCE_DIR}/include
        $ENV{VCPKG_ROOT}/installed/x64-windows/include
        $ENV{VCPKG_ROOT}/installed/x86-windows/include
        NO_DEFAULT_PATH
    )
    
    if(NOT NLOHMANN_JSON_INCLUDE_DIR)
        message(FATAL_ERROR "nlohmann/json not found. Please install via vcpkg: vcpkg install nlohmann-json")
    endif()
    
    include_directories(${NLOHMANN_JSON_INCLUDE_DIR})
else()
    find_package(nlohmann_json REQUIRED)
endif()

# Set Boost include directories
if(Boost_FOUND)
    include_directories(${Boost_INCLUDE_DIRS})
endif()

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/include)
include_directories(${CMAKE_SOURCE_DIR}/external)

# Add subdirectories
add_subdirectory(src)
add_subdirectory(tests)

# Main executable
add_executable(solana_arbitrage_bot
    src/main.cpp
    src/blockchain_adapters.cpp
    src/market_data_provider.cpp
    src/arbitrage_engine.cpp
    src/order_manager.cpp
    src/risk_manager.cpp
    src/config_manager.cpp
    src/logger.cpp
)

# Windows-specific source files
if(WIN32)
    target_sources(solana_arbitrage_bot PRIVATE
        src/windows_compat.cpp
    )
endif()

# Link libraries
target_link_libraries(solana_arbitrage_bot
    PRIVATE
    Boost::system
    Boost::thread
    OpenSSL::SSL
    OpenSSL::Crypto
    Threads::Threads
    blockchain_adapters
    market_data_provider
    arbitrage_engine
    order_manager
    risk_manager
    config_manager
    logger
)

# Windows-specific libraries
if(WIN32)
    target_link_libraries(solana_arbitrage_bot PRIVATE
        ws2_32
        iphlpapi
        crypt32
    )
endif()

# Install
install(TARGETS solana_arbitrage_bot DESTINATION bin)

# Windows compatibility test
if(WIN32)
    add_executable(windows_compatibility_test
        tests/test_windows_compatibility.cpp
    )
    
    target_link_libraries(windows_compatibility_test
        PRIVATE
        Boost::system
        Boost::thread
        OpenSSL::SSL
        OpenSSL::Crypto
        Threads::Threads
    )
    
    if(WIN32)
        target_link_libraries(windows_compatibility_test PRIVATE
            ws2_32
            iphlpapi
            crypt32
        )
    endif()
endif()
